#include <fstream>
using namespace std;


void EditArray(double** a, int n)
{
	int i, j, k;

	for (i = 0; i < n / 2; i++)
	{
		for (j  = 0; j < (n/2)+(n%2); j++)
		{
			k = *(*(a + j) + n - 1 - i);
			*(*(a + j) + n - 1 - i) = *(*(a + i) + j);
			*(*(a + i) + j) = *(*(a + n - 1 - j) + i);
			*(*(a + n - 1 - j) + i) = *(*(a + n - 1 - i) + n - 1 - j);
			*(*(a + n - 1 - i) + n - 1 - j) = k;
		}
	}
}

void RealSize(fstream& f, int& n)
{
	int i = 0;
	int j = 0;
	double x;
	char y;

	f.open("input.txt", ios::in);
	if (f.bad() || f.peek() == ifstream::traits_type::eof())
	{
		n = 0;
		f.close();
	}
	else
	{
		while (true)
		{
			f >> x;
			j++;
			f.unsetf(ios::skipws);
			f >> y;
			if ((y == '\n') || f.eof())
			{
				i++;

				if (i == 1)
				{
					n = j;
				}

				if (i > n)
				{
					f.setf(ios::skipws);
					break;
				}

				if ((i == n) && (j < n))
				{
					n = i - 1;
					f.setf(ios::skipws);
					break;
				}

				if (j < n)
				{
					n = j;
				}
				j = 0;

				if (f.eof())
				{
					if (i < n)
					{
						n = i;
					}
					f.setf(ios::skipws);
					break;
				}
			}
			f.setf(ios::skipws);
		}
		f.close();
	}
}

void ReadArray(fstream& f, double** a, int n)
{
	f.open("input.txt", ios::in);
	int i, j;
	double x;
	char y;

	for (i = 0; i < n; i++)
	{
		for (j = 0; j < n; j++)
		{
			f >> x;
			*(*(a + i) + j) = x;
			if (j == n - 1)
			{
				f.unsetf(ios::skipws);
				while (true)
				{
					f >> y;
					if ((y == '\n') || (f.eof()))
					{
						f.setf(ios::skipws);
						break;
					}
				}

			}

		}

	}

}

void PrintArray(fstream& f, double** a, int n)
{
	int i, j;
	for (i = 0; i < n; i++)
	{
		for (j = 0; j < n; j++)
		{
			cout << *(*(a + i) + j) << " ";
			f << *(*(a + i) + j) << " ";
		}
		cout << "\n";
		f << "\n";
	}
	cout << "\n";
	f << "\n";
}

void DeleteArray(double** a, int n)
{
	int i, j;
	for (i = 0; i < n; i++)
	{
		delete[] *(a + i);
	}
	delete[] a;
}
